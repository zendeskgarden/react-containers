## Controlled

<Canvas>
  <Story
    name="Controlled"
    args={{
      selectedItem: '0'
    }}
    argTypes={{ defaultSelectedIndex: { control: false } }}

    {args => {
      const updateArgs = useArgs()[1];
      const handleSelect = selectedItem => updateArgs({ selectedItem });
      return <TabsStory {...args} onSelect={handleSelect} />;
    }}
  </Story>
