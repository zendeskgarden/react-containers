## Uncontrolled

## Controlled

<Canvas>
  <Story
    name="Controlled"
    args={{
      selectedItem: 'item-3'
    }}
    argTypes={{
      selectedItem: { control: { type: 'text' } },
      focusedItem: { control: { type: 'text' } },
      defaultFocusedItem: { control: false },
      defaultSelectedItem: { control: false }
    }}
  >
    {args => {
      const updateArgs = useArgs()[1];
      const handleSelect = selectedItem => updateArgs({ selectedItem });
      const handleFocus = focusedItem => updateArgs({ focusedItem });
      return <SelectionStory {...args} onSelect={handleSelect} onFocus={handleFocus} />;
    }}
  </Story>
</Canvas>
