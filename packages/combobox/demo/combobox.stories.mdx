## Controlled

<Canvas>
  <Story
    name="Controlled"
    args={{ isExpanded: false, inputValue: '', activeIndex: -1, selectionValue: null }}
    argTypes={{ defaultExpanded: { control: false }, defaultSelectionValue: { control: false } }}
  >
    {args => {
      const updateArgs = useArgs()[1];
      const handleChange = changes => {
        const state = {
          ...(changes.isExpanded !== undefined && { isExpanded: changes.isExpanded }),
          ...(changes.selectionValue !== undefined && { selectionValue: changes.selectionValue }),
          ...(changes.inputValue !== undefined && { inputValue: changes.inputValue }),
          ...(changes.activeIndex !== undefined && { activeIndex: changes.activeIndex })
        };
        updateArgs(state);
      };
      return <ComboboxStory {...args} onChange={handleChange} />;
    }}
  </Story>
</Canvas>
