# Demo

## Uncontrolled

## Controlled

<Canvas>
  <Story
    name="Controlled"
    args={{ valueNow: 200 }}
    argTypes={{ defaultValueNow: { control: false } }}
  >
    {({ ...args }) => {
      const updateArgs = useArgs()[1];
      const handleChange = valueNow => updateArgs({ valueNow });
      return <SplitterStory {...args} onChange={handleChange} />;
    }}
  </Story>
</Canvas>
